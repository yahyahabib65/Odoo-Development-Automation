**renderer.py changes:**

Modify `render_module()` to emit artifact state transitions. The state tracking is wrapped in try/except so it NEVER blocks generation (OBS-01 requirement).

At the top of render_module(), after `module_dir` is set and `created_files` is initialized, add state initialization:
```python
# --- Artifact state tracking (OBS-01) ---
try:
    from odoo_gen_utils.artifact_state import (
        ArtifactKind, ArtifactStatus, ModuleState, save_state,
    )
    _state = ModuleState(module_name=module_name)
except Exception:
    _state = None
```

After each major artifact render section, add state transition calls wrapped in try/except. The pattern:
```python
if _state is not None:
    try:
        _state = _state.transition(
            kind=ArtifactKind.MODEL.value,
            name=model["name"],
            file_path=str(created_files[-1].relative_to(module_dir)),
            new_status=ArtifactStatus.GENERATED.value,
        )
    except Exception:
        pass  # State tracking must never block generation
```

Add state transitions for these artifact types:
- After manifest render: kind=MANIFEST, name="__manifest__", file_path="__manifest__.py"
- After each model file render: kind=MODEL, name=model["name"], file_path=relative path
- After each view file render (per model): kind=VIEW, name=model["name"], file_path=relative path
- After security files render: kind=SECURITY, name="ir.model.access.csv", file_path="security/ir.model.access.csv"
- After each test file render: kind=TEST, name=model["name"], file_path=relative path

At the end of render_module(), before the return statement, save state:
```python
if _state is not None:
    try:
        save_state(_state, module_dir)
    except Exception:
        pass  # State tracking must never block generation
```

Do NOT change the function signature or return type. Do NOT add state as a parameter. The state tracking is an internal side effect.

```python
class TestRenderModuleStateIntegration:
    """Integration tests: render_module emits artifact states."""

    def test_render_module_creates_state_file(self, tmp_path):
        """render_module() creates .odoo-gen-state.json sidecar."""
        # Minimal spec with one model
        spec = {
            "module_name": "test_state_mod",
            "models": [{"name": "test.model", "fields": []}],
        }
        from odoo_gen_utils.renderer import render_module, get_template_dir
        template_dir = get_template_dir()
        files, warnings = render_module(spec, template_dir, tmp_path)
        state_file = tmp_path / "test_state_mod" / ".odoo-gen-state.json"
        assert state_file.exists(), "State sidecar file should be created"
        data = json.loads(state_file.read_text())
        assert data["module_name"] == "test_state_mod"
        assert len(data["artifacts"]) > 0
        # Check at least model and manifest artifacts tracked
        kinds = {a["kind"] for a in data["artifacts"]}
        assert "model" in kinds
        assert "manifest" in kinds

    def test_render_module_state_failure_does_not_block(self, tmp_path, monkeypatch):
        """If state tracking fails, render_module still succeeds."""
        # Monkey-patch ModuleState.transition to raise
        import odoo_gen_utils.artifact_state as ast_mod
        original_transition = ast_mod.ModuleState.transition
        def _exploding_transition(self, *args, **kwargs):
            raise RuntimeError("State tracking exploded")
        monkeypatch.setattr(ast_mod.ModuleState, "transition", _exploding_transition)

        spec = {
            "module_name": "test_no_block",
            "models": [{"name": "test.model", "fields": []}],
        }
        from odoo_gen_utils.renderer import render_module, get_template_dir
        template_dir = get_template_dir()
        # Must not raise
        files, warnings = render_module(spec, template_dir, tmp_path)
        assert len(files) > 0, "Module files should still be created"
```
