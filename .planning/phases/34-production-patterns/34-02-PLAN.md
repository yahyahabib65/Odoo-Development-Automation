       from odoo import api, fields, models
       from dateutil.relativedelta import relativedelta


       class {{ wizard_class }}(models.TransientModel):
           _name = "{{ wizard.name }}"
           _description = "{{ wizard.name | replace('.', ' ') | title }}"

               cutoff = fields.Date.today() - relativedelta(days=self.days_threshold)
               records = self.env['{{ wizard.target_model }}'].search([
                   ('active', '=', True),
                   ('create_date', '<', cutoff),
               ])
               records.write({'active': False})
               return {'type': 'ir.actions.act_window_close'}
       ```

       {% if is_archival %}
           @api.model
           def _cron_archive_old_records(self):
               """Scheduled action: archive records older than threshold.

               Uses batch processing with commit-per-batch to avoid long transactions.
               """
               from dateutil.relativedelta import relativedelta
               BATCH_SIZE = {{ archival_batch_size }}
               cutoff = fields.Date.today() - relativedelta(days=int(
                   self.env['ir.config_parameter'].sudo().get_param(
                       '{{ module_name }}.archival_days', default={{ archival_days }}
                   )
               ))
               while True:
                   records = self.search([
                       ('active', '=', True),
                       ('create_date', '<', cutoff),
                   ], limit=BATCH_SIZE)
                   if not records:
                       break
                   records.write({'active': False})
                   self.env.cr.commit()
       {% endif %}
       ```
